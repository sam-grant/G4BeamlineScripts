#param Wedge_pbar_upperWidth=.001

#param Wedge_pbar_lowerWidth=0.0001

#     material=KAPTON \
#     material=Be \

     height=$Wedge_pbar_height \
     upperWidth=$Wedge_pbar_upperWidth \
     lowerWidth=$Wedge_pbar_lowerWidth \
     Xul=-175. \
     Xur=175. \
     Xll=-175. \
     Xlr=175. \
     length=$Wedge_pbar_length \
     maxStep=50.e-3\
     material=Be \
     color=$Red \
     kill=0

      place Wedge_pbar \
            x=$Wedge_pbar_x \
            y=$Wedge_pbar_y \
            z=$Wedge_pbar_z
